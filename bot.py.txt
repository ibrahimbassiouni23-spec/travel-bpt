users = {}

# Start
async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    keyboard = [["احجز ميعاد 🚀", "تواصل واتساب 📲"]]
    reply_markup = ReplyKeyboardMarkup(keyboard, resize_keyboard=True)

    await update.message.reply_text(
        "أهلاً بيك 👋\nاحجز ميعادك بسهولة في أقل من دقيقة 🚀\n\n⚠️ المواعيد محدودة اليوم",
        reply_markup=reply_markup
    )

    users[update.effective_user.id] = {}

# Messages
async def handle_message(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user_id = update.effective_user.id
    text = update.message.text

    if user_id not in users:
        users[user_id] = {}

    user = users[user_id]

    # واتساب مباشر
    if text == "تواصل واتساب 📲":
        link = "https://wa.me/201032013792"
        keyboard = [[InlineKeyboardButton("فتح واتساب 📲", url=link)]]

    if text == "احجز ميعاد 🚀":
        keyboard = [["نعم", "لا"]]

    if "passport" not in user:

    # إيطاليا تفاصيل

{wa_link}
